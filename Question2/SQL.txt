CREATE TABLE Product(
   id INTEGER PRIMARY KEY AUTOINCREMENT,
   name TEXT NOT NULL
);

CREATE TABLE Category(
   id INTEGER PRIMARY KEY AUTOINCREMENT,
   name TEXT NOT NULL
);

CREATE TABLE ProductToCategory(
   id INTEGER PRIMARY KEY AUTOINCREMENT,
   productId INTEGER,
   categoryId INTEGER
);
